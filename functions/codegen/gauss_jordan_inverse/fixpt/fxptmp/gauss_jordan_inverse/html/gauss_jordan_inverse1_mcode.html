1function [A_inv] = gauss_jordan_inverse(A,size_p)
 2%GAUSS_JORDAN_INVERSE Summary of this function goes here
 3% This function implements the Gauss-Jordan method for calculating inverse of a square matrix.
 4% It acts as a high level model for later implementation in hardware.
 5%   Detailed explanation goes here
 6% USAGE:
 7% Inputs:
 8% A         -   Matrix of size p x p
 9% size_p    -   column size
10% Outputs:
11% A_inv     -   inverse matrix
12
13A_inv = eye(size_p);
14
15% Forward elimination to build an upper triangular matrix
16for (i=1:1:size_p)
17   if (A(i,i) == 0)
18      for (j =i+1:1:size_p)
19          if (A(j,j)~=0)
20              % The operations below will be different in hardware, because
21              % of parallell operations
22              %temp_i = row(i);
23              %row(i) = row(j);
24              %row(j) = temp_i;
25              temp_i = A(i,:);
26              A(i,:) = A(j,:);
27              A(j,:) = temp_i;
28          end
29      end 
30   end
31   if (A(i,i) ==0)
32      error('Matrix is singular'); 
33   end
34   for (j = i +1:1: size_p)
35        % The operations below will be different in hardware, because
36        % of parallell operations
37        A_j_i_temp =A(j,i);
38        A_i_i_temp = A(i,i);
39        A(j,:) = A(j,:)- A(i,:)*A(j,i)/A(i,i);
40        A_inv(j,:) = A_inv(j,:) - A_inv(i,:)*A_j_i_temp/A_i_i_temp;
41   end
42end
43%triangular_matrix = A_inv;
44
45% Backward elimination to build a diagonal matrix
46for(i=size_p:-1:2)
47   for( j=i-1:-1: 1)
48        % The operations below will be different in hardware, because
49        % of parallell operations
50        A_j_i_temp =A(j,i);
51        A_i_i_temp = A(i,i);
52        A(j,:) = A(j,:)-A(i,:)*A(j,i)/A(i,i);
53        A_inv(j,:) = A_inv(j,:) - A_inv(i,:)*A_j_i_temp/A_i_i_temp;
54   end
55end
56
57% Last division to build an identity matrix
58for ( i = 1:+1:size_p)
59    A_inv(i,:)= A_inv(i,:)*1/A(i,i);
60end
61
62
63
64end
